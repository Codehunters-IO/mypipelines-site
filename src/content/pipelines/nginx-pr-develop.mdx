# Runs: commit-lint + build (image + nginx -t) + smoke test (/__health)

  pull_request:
    branches:
      - develop

  validate:
    uses: <org>/ci-templates/.github/workflows/nginx-main-pipeline.yml@main
    with:
      run_commit_lint: true
      run_build: true
      run_test: true
